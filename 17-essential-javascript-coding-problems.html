4. Prime Check

Problem: Implementation of Prime Check algorithm in JavaScript.

Run Live 
function isPrime(n) {
  if (n < 2) return false;
  for(let i=2; i<=Math.sqrt(n); i++) if(n%i === 0) return false;
  return true;
}
console.log(isPrime(19));

Efficient prime check iterating up to the square root.

Time Complexity

O(sqrt(n))

Space Complexity

O(1)

6. Binary Search

Problem: Implementation of Binary Search algorithm in JavaScript.

Run Live 
function bs(arr, x) {
  let l = 0, r = arr.length - 1;
  while(l <= r) {
    let m = Math.floor((l+r)/2);
    if(arr[m] === x) return m;
    if(arr[m] < x) l = m + 1; else r = m - 1;
  }
  return -1;
}
console.log(bs([1,2,3,4,5], 4));

Classic binary search implementation for sorted arrays.

Time Complexity

O(log n)

Space Complexity

O(1)

14. Armstrong Number

Problem: Implementation of Armstrong Number algorithm in JavaScript.

Run Live 
function isArm(n) {
  let s = n.toString();
  let p = s.length;
  let sum = [...s].reduce((a, d) => a + Math.pow(parseInt(d), p), 0);
  return sum === n;
}
console.log(isArm(153));

Calculates if sum of digits^power equals number.

Time Complexity

O(d)

Space Complexity

O(d)
